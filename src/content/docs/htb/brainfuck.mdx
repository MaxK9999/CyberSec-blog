This means we'll have to decrypt it using a `python script` I found online:

```python
import re

def egcd(a, b):
    if a == 0:
        return (b, 0, 1)
    else:
        g, y, x = egcd(b % a, a)
        return (g, x - (b // a) * y, y)

def modinv(a, m):
    g, x, y = egcd(a, m)
    if g != 1:
        raise Exception('modular inverse does not exist')
    else:
        return x % m

debug = open("debug.txt","r")
enc_pass = open("output.txt","r")

arr=[]  #arr[0]=p, arr[1]=q, arr[2]=e

for line in debug.readlines():
    line = line.strip()
    arr.append(int(line))
p,q,e = arr

c = re.findall("\d+",enc_pass.read())
c = int(c[0])

n=p*q
tot = (p-1)*(q-1)
d = modinv(e,tot)
m = pow(c,d,n)

root = hex(m)[2:-1].decode('hex')

print root
```
